13.$ unzip opencv_contrib.zip
14.$ cd ~/opencv-4.4.0/
15.$ mkdir build
16.$ cd build

18.$ -D CMAKE_INSTALL_PREFIX=/usr/local \
19.$ -D INSTALL_PYTHON_EXAMPLES=ON \
20.$ -D OPENCV_EXTRA_MODULES_PATH=~/opencv_contrib-4.4.0/modules \
